WBStartup:freest 0:ScreensBitMap 0,0

NEWTYPE .STAR
	X.q
	OLDX.q
	Y.q
	SPEED.q
	FLAG.q
End NEWTYPE

#MAXSTARS=20

Dim List MYSTAR.STAR(#MAXSTARS)

.START:

localstat 0,0,0,100,11,$E,"STARS",2,1

##[$80CA] MYSTAR()

Repeat

If globalloc(MYSTAR())
	\X=WBWidth;-Int(Rnd(40))
	\Y=Int(Rnd(WBHeight))
	\SPEED=(Int(Rnd(3))+1)*5
EndIf

Gosub MOVESTARS

;VWait

Gosub DRAWSTARS

EV.l=Event

Until EV=$200

End


.MOVESTARS:

newmem MYSTAR()

While globfree(MYSTAR())

	If \X-\SPEED<0
		If \FLAG=True
			Point \X,\Y,0
		EndIf
		freelast MYSTAR()
	Else
		\OLDX=\X
		\X-\SPEED
	EndIf

;	Gosub DRAWSTARS
Wend

Return

.DRAWSTARS:

newmem MYSTAR()

While globfree(MYSTAR())

;	VWait

	If \FLAG=True
		Point \OLDX,\Y,0
	EndIf

;	VWait

	If Cls(\X,\Y)=0
		Point \X,\Y,1
		\FLAG=True
	Else
		\FLAG=False
	EndIf

Wend

Return
