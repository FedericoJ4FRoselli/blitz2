Function.w Valid{cur.w,pl.w}
	Shared numplyr.w,np()

	v=On

	If cur<>6		;just one thingee
		If numplyr>3
			v=Off
		Else
			Select cur
				Case 0:v=On										;joy0
				Case 1:If numplyr<1 Then v=Off ;joy2
				Case 2:v=On										 ;arrows
				Case 3:v=On										 ;ctrl ?
				Case 4:If numplyr<2 Then v=Off ;keypad
				Case 5:v=Off ; serial
				Case 6:v=On ; computer
				Case 7:v=Off ; <shrugs>

			End Select
		EndIf

		If v
			For npl=0 To 7
				If (npl<>pl) AND (np(npl)\p>-1)
					If ##[$BA8B](4,5+npl*2)=cur Then v=Off
				EndIf
			Next
		EndIf
	EndIf

	Function Return v
End Function

.PruneCar
	If ##[$BA8B](2)
		ttcars.w=0
		For roll.w=0 To maxcar
			happy=0
			For pitch.w=0 To maxpl
				If np(pitch)\car=roll
					If np(pitch)\p=-1
						If happy=0 Then happy=1
					Else
						happy=2
					EndIf
				EndIf
			Next
			If happy<>0 Then ttcars+1
			cvh(roll)\ndd=happy
		Next

		While ttcars>avcars
			killer.w=-1
			For roll.w=0 To maxcar
				If roll<>exempt
					If cvh(roll)\ndd=2 Then killer=roll
				EndIf
			Next
			For roll.w=0 To maxcar
				If roll<>exempt
					If cvh(roll)\ndd=1 Then killer=roll
				EndIf
			Next

			cvh(killer)\ndd=0
			dest.w=killer
			Repeat
				dest.w=QWrap(dest.w+1,0,maxcar+1)
			Until cvh(dest)\ndd>0
			For roll.w=0 To maxpl
				If np(roll)\car=killer Then np(roll)\car=dest
			Next
			ttcars-1
		Wend
	Else
		For roll.w=0 To maxpl
			np(roll)\car=exempt
		Next
		For roll.w=0 To maxcar
			cvh(roll)\ndd=0
		Next
		cvh(exempt)\ndd=2
	EndIf

		PlotCar{cnt}
	Next
