+; A simple ball swarming proggie
; J.A.Eames 6/1/94
; Manchester is as wet as my thighs!!!

;load in ball shape
;
LoadShapes 0,"code:ball.shapes"

;use blitz mode
;
BLITZ

;turn error checking off
;
##[$80D3]

;setup a bitmap
;
lmaxlen 0,320,200,3
lmaxlen 1,320,200,3

;setup a slice for it
;
Slice 0,44,320,200,$fff8,3,8,32,320,320

;show it
;
Show 0

;turn mouse reading on
;
Mouse On

;setup a type for the balls
;
NEWTYPE .coords
	x.w
	y.w
	inertx.b
	inerty.b
	pic.b
End NEWTYPE

;number of balls
;
maxm=64	; CHANGE ME FOR MORE BALLS!!!

;keep track of current screen
;
scr=0

;cycle counter and swarm flag
;
cou=0
swarm=0
inertmax=12
switch=10000	; LENGTH OF SWARM AND BOUNCE MODES


;define space for maxm balls
;
Dim balls.coords(maxm)

;fill them with random coordinates
;
For n=1 To maxm
	balls(n)\x=Rnd(280)+20,Rnd(160)+20,Rnd(5)+1,Rnd(5)+1,Rnd(3)
;	balls(n)\x=Rnd(280)+20,Rnd(160)+20,1,1
Next n

longmult 0,maxm
longmult 1,maxm

While Joyb(0)=0

	VWait
	Show scr
	scr=1-scr
	Use lmaxlen scr
	UnQueue scr
	lx=balls(1)\x
	ly=balls(1)\y
	For n=1 To maxm
		x=balls(n)\x
		y=balls(n)\y
		inertx=balls(n)\inertx
		inerty=balls(n)\inerty
		cou=cou+1
		If cou=switch
			swarm=1-swarm
			cou=0
			If swarm=0
				For m=1 To maxm
					balls(m)\inertx=Rnd(5)+1,Rnd(5)+1
				Next m
			EndIf
		EndIf

		If swarm=1
		If x>lx AND inertx>=-inertmax
			inertx=inertx-1
		EndIf
		If x<lx AND inertx<=inertmax
			inertx=inertx+1
		EndIf
		If y>ly AND inertx>=-inertmax
			inerty=inerty-1
		EndIf
		If y<ly AND inerty<=inertmax
			inerty=inerty+1
		EndIf
		EndIf

		x=x+inertx
		y=y+inerty
		If x>=310
			x=310
			inertx=-inertx
		EndIf
		If x<=10
			x=10
			inertx=-inertx
		EndIf
		If y>=190
			y=190
			inerty=-inerty
		EndIf
		If y<=10
			y=10
			inerty=-inerty
		EndIf
		QBlit scr,balls(n)\pic,balls(n)\x,balls(n)\y
		balls(n)\x=x
		balls(n)\y=y
		balls(n)\inertx=inertx
		balls(n)\inerty=inerty
	Next n
;	MOVE #-1,$dff180
Wend

;turn error checking back on
;
##[$80D2]

End
